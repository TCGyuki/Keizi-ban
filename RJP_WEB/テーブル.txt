CREATE TABLE Thread (
	Thread_ID       varchar2(8),
	Thread_Title    varchar2(100) DEFAULT 'テーマ無し',
	Thread_Date     date DEFAULT SYSDATE,
	Thread_User　	varchar2(20) DEFAULT '名もなき人間',
	CONSTRAINT PK_Thread PRIMARY KEY(Thread_ID)
);


	

CREATE TABLE Response(
	Thread_ID 	varchar2(8),
	Res_ID　	varchar2(8),
	Res_User　	varchar2(20) DEFAULT '名もなき人間',
	Res_Date 	date DEFAULT SYSDATE,
	Res_comment 	varchar2(500) NOT NULL,
	CONSTRAINT PK_ResID PRIMARY KEY(Res_ID),
	CONSTRAINT FK_Thread_ID FOREIGN KEY(Thread_ID) REFERENCES Thread(Thread_ID)
);
